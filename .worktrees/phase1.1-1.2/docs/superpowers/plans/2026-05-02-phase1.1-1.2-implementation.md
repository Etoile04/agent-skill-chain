- [ ] **Step 3: 实现 validate-strategy-hint.sh**

```bash
cat > scripts/validate-strategy-hint.sh << 'SCRIPT'
#!/usr/bin/env bash
# strategy_hint 校验：检查长度、文件路径、CLI 命令、错误代码、密钥
set -euo pipefail
INPUT="${1:?Usage: validate-strategy-hint.sh <stepplan.json>}"
MAX_LEN=200
PASS=0; FAIL=0; WARN=0

# 从 JSON 提取所有 strategy_hint
hints=$(python3 -c "
import json,sys
d=json.load(open(sys.argv[1]))
for s in d.get('steps',[]):
    h=s.get('strategy_hint','')
    if h: print(f'{s.get(\"id\",\"?\")}|{h}')
" "$INPUT" 2>/dev/null || echo "")

while IFS='|' read -r sid hint; do
  [[ -z "$hint" ]] && continue
  issues=()

  # 长度检查
  hint_len=${#hint}
  if (( hint_len > MAX_LEN )); then
    issues+=("too long ($hint_len > $MAX_LEN)")
  fi

  # 文件路径检查
  if echo "$hint" | grep -qE '/(home|tmp|var|etc|usr|opt|Users)/|C:\\'; then
    issues+=("contains file path")
  fi

  # CLI 命令检查
  if echo "$hint" | grep -qE '\b(npm|yarn|pip|cargo|go|rustc|gcc|make|cmake|git|curl|wget|docker|kubectl|bash|sh|zsh)\b'; then
    issues+=("contains CLI command name")
  fi

  # 错误代码检查
  if echo "$hint" | grep -qE 'ERR-[0-9]|HTTP [0-9]{3}'; then
    issues+=("contains error code")
  fi

  # 密钥检查
  if echo "$hint" | grep -qiE '(api.key|token|secret|password|credential)'; then
    issues+=("contains sensitive keyword")
  fi

  if (( ${#issues[@]} == 0 )); then
    echo "PASS: step[$sid] — OK (len=$hint_len)"
    ((PASS++))
  else
    echo "FAIL: step[$sid] — ${issues[*]}"
    ((FAIL++))
  fi
done <<< "$hints"

echo "--- Results: $PASS passed, $FAIL failed ---"
[[ $FAIL -eq 0 ]]
SCRIPT
chmod +x scripts/validate-strategy-hint.sh
```
